📱 Ejemplos de Integración

JavaScript / React Native

// Hook personalizado para React
import { useState, useEffect } from 'react';

const PROMOS_URL = 'https://raw.githubusercontent.com/TU-USUARIO/promos-argentina/main/data/promos.json';

export function usePromos() {
  const [promos, setPromos] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetch(PROMOS_URL)
      .then(res => res.json())
      .then(data => {
        setPromos(data.promociones);
        setLoading(false);
      })
      .catch(err => {
        setError(err);
        setLoading(false);
      });
  }, []);

  return { promos, loading, error };
}

// Funciones de filtrado
export const filtrarPorComercio = (promos, comercio) => {
  return promos.filter(p => 
    p.comercio.toLowerCase() === comercio.toLowerCase()
  );
};

export const topPromos = (promos, limit = 10) => {
  return [...promos]
    .sort((a, b) => (b.score || 0) - (a.score || 0))
    .slice(0, limit);
};

Python / Django / Flask

import requests
from datetime import datetime

class PromosAPI:
    BASE_URL = "https://raw.githubusercontent.com/TU-USUARIO/promos-argentina/main/data"
    
    @staticmethod
    def obtener_promos():
        """Obtiene todas las promociones"""
        response = requests.get(f"{PromosAPI.BASE_URL}/promos.json")
        response.raise_for_status()
        data = response.json()
        return data['promociones']
    
    @staticmethod
    def top_promos(promos, limit=10):
        """Obtiene las mejores promociones por score"""
        sorted_promos = sorted(promos, key=lambda x: x.get('score', 0), reverse=True)
        return sorted_promos[:limit]

# Uso
api = PromosAPI()
promos = api.obtener_promos()
top_5 = api.top_promos(promos, 5)

for i, promo in enumerate(top_5, 1):
    print(f"{i}. {promo['comercio']}: {promo['beneficio']} (Score: {promo['score']})")

Flutter / Dart

import 'dart:convert';
import 'package:http/http.dart' as http;

class PromosService {
  static const String baseUrl = 
    'https://raw.githubusercontent.com/TU-USUARIO/promos-argentina/main/data';

  Future<List<Promo>> fetchPromos() async {
    final response = await http.get(Uri.parse('$baseUrl/promos.json'));

    if (response.statusCode == 200) {
      final data = json.decode(response.body);
      return (data['promociones'] as List)
          .map((p) => Promo.fromJson(p))
          .toList();
    } else {
      throw Exception('Error al cargar promociones');
    }
  }
}

Documentación de Campos

Campo Tipo Descripción Ejemplo
id string Hash único MD5 (12 chars) "a3f7b2c91d4e"
comercio string Nombre normalizado del supermercado "Carrefour"
banco string Banco emisor (normalizado) "Banco Galicia"
metodo_pago array Métodos de pago aceptados ["Visa", "Mastercard"]
beneficio string Descuento principal "20%", "3x2"
tope string | null Tope de reintegro "$5.000" o null
dias array Días válidos ([] = todos) ["Lunes", "Martes"]
score number Puntuación de calidad (0-100) 85
destacada boolean true si score ≥ 70 true
confianza number Nivel anti-fraude (0-100) 100

🚀 Instalación Local

Prerequisitos: Python 3.11+, pip, Git 
# 1. Clonar el repositorio
git clone https://github.com/TU-USUARIO/promos-argentina.git
cd promos-argentina

# 2. Crear entorno virtual
python -m venv venv
source venv/bin/activate  # Linux/Mac
# venv\Scripts\activate   # Windows

# 3. Instalar dependencias
pip install -r requirements.txt

# 4. Instalar Playwright
playwright install chromium

# 5. Ejecutar scraping
python scripts/main.py

# 6. Verificar archivos generados
ls -lh data/

⚙️ GitHub Actions - Automatización

Configuración Importante:
  1. Ve a Settings → Actions → General
  2. Selecciona "Read and write permissions"
  3. Marca "Allow GitHub Actions to create and approve pull requests"
  4. Guarda cambios

Ejecución Automática: Todos los días a las 6:00 AM (hora Argentina)

Ejecución Manual:

  1. Ve a la pestaña "Actions"
  2. Selecciona "Actualizar Promociones Diarias"
  3. Click en "Run workflow"
  4. Confirma la ejecución
